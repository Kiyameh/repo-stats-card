üöÄ Quick Start

Installation

npm install github-repo-stats-card

Basic Usage

import { getRepoStats, renderRepoCard } from 'github-repo-stats-card'

// Get repository data
const stats = await getRepoStats('facebook/react')

// Render the card
renderRepoCard('#my-card', stats)

With Authentication

const stats = await getRepoStats(
  'facebook/react', 
  'ghp_your_token_here'
)

Multiple Repositories

const [reactStats, vueStats] = await Promise.all([
  getRepoStats('facebook/react'),
  getRepoStats('vuejs/vue')
])

renderRepoCard('#react-card', reactStats)
renderRepoCard('#vue-card', vueStats)
